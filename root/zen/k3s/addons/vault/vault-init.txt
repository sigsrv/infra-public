kubectl exec -n vault vault-0 -- \
    vault operator init \
    -pgp-keys="keybase:ecmaxp" \
    -root-token-pgp-key="keybase:ecmaxp" \
    -key-shares=1 \
    -key-threshold=1

for i in (seq 0 2)
    kubectl exec --context k3s -n vault vault-$i -- \
        vault operator unseal \
        (op read "op://Personal/sigsrv-k3s-vault/encrypted/unseal-key" | base64 -d | gpg --decrypt)
end

for i in (seq 0 2)
    lxc exec --project sigsrv-k3s sigsrv-k3s-master-$i -- sh -c "sudo apt update && sudo apt upgrade -y"
    lxc exec --project sigsrv-k3s sigsrv-k3s-worker-$i -- sh -c "sudo apt update && sudo apt upgrade -y"
end
